import re

sql_query = """
SELECT column1, column2
FROM table1
WHERE column3 = (SELECT column4 FROM table2 WHERE condition)
"""

pattern = r'\((SELECT .*?FROM .*?WHERE .*?\)|SELECT .*?FROM .*?WHERE .*?\))'

matches = re.findall(pattern, sql_query, re.IGNORECASE | re.DOTALL)

for match in matches:
    print(match.strip())
